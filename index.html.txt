<script>
function goTo(id) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  const s = document.getElementById(id);
  if (s) { s.classList.add('active'); s.scrollTop = 0; }
}

function navTo(screenId, navId) {
  goTo(screenId);
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
  const n = document.getElementById(navId);
  if (n) n.classList.add('active');
}

let toastT;
function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  clearTimeout(toastT);
  toastT = setTimeout(() => t.classList.remove('show'), 2400);
}

function selectFood(el) {
  document.querySelectorAll('.food-item').forEach(f => f.classList.remove('selected'));
  el.classList.add('selected');
}

function updateSlider(input) {
  const v = input.value, p = ((v-1)/(50-1))*100;
  input.style.background = `linear-gradient(to right,var(--gold) 0%,var(--gold) ${p}%,#e0e0e0 ${p}%,#e0e0e0 100%)`;
  const b = document.getElementById('qty-bubble');
  if (b) { b.textContent = v+' كجم'; b.style.left = `calc(${p}% - 26px)`; }
}

function submitDonation() {
  showToast('✅ تم إرسال التبرع بنجاح!');
  setTimeout(() => goTo('screen-success'), 700);
}

function assignVolunteer(name) {
  showToast('✅ تم تعيين ' + name + ' بنجاح!');
  setTimeout(() => goTo('screen-tracking'), 800);
}

document.addEventListener('DOMContentLoaded', () => {
  const s = document.getElementById('qty-slider');
  if (s) updateSlider(s);
});
</script>
